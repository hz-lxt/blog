一、发送邮件

import smtplib
from email.mime.text import MIMEText
from email.header import Header
msg_from = '572478035@qq.com'  # 发送方邮箱
passwd = 'ffaoofjltdfmbcic'  # 填入发送方邮箱的授权码(填入自己的授权码，相当于邮箱密码)
msg_to = ['798818871@qq.com','904925371@qq.com']  # 收件人邮箱
# msg_to = '616564099@qq.com'  # 收件人邮箱

subject = "邮件接口测试"  # 主题
content = "邮件内容，我是邮件内容，哈哈哈"
# 生成一个MIMEText对象（还有一些其它参数）
msg = MIMEText(content)
# 放入邮件主题
msg['Subject'] = subject
# 也可以这样传参
# msg['Subject'] = Header(subject, 'utf-8')
# 放入发件人
msg['From'] = msg_from
# 放入收件人
# msg['To'] = '616564099@qq.com'
# msg['To'] = '发给你的邮件啊'
try:
    # 通过ssl方式发送，服务器地址，端口
    s = smtplib.SMTP_SSL("smtp.qq.com", 465)
    # 登录到邮箱
    s.login(msg_from, passwd)
    # 发送邮件：发送方，收件方，要发送的消息
    s.sendmail(msg_from, msg_to, msg.as_string())
    print('成功')
except Exception as e:
    print(e)
finally:
    s.quit()

二、生成二维码

1 生成普通二维码

from MyQR import myqr  #注意大小写
myqr.run(words="网址链接或者字符串，不支持中文")

# 会在当前路径生成一个“qrcode.png”文件，扫描二维码就会跳转到指定路径

2 生成带图片的二维码

from MyQR import myqr
myqr.run(
    words='Why can Li Xiaotong be so beautiful',
    # 扫描二维码后，显示的内容，或是跳转的链接
    version=5,  # 设置容错率
    level='H',  # 控制纠错水平，范围是L、M、Q、H，从左到右依次升高
    picture='a.jpg',  # 图片所在目录，可以是动图
    colorized=True,  # 黑白(False)还是彩色(True)
    contrast=1.0,  # 用以调节图片的对比度，1.0 表示原始图片。默认为1.0。
    brightness=1.0,  # 用来调节图片的亮度，用法同上。
    save_name='二维码.png',  # 控制输出文件名，格式可以是 .jpg， .png ，.bmp ，.gif
)

3 支持汉字生成二维码

import qrcode
qr = qrcode.QRCode(version=5,
                          error_correction=qrcode.constants.ERROR_CORRECT_L,
                          box_size=8,
                          border=4,
                          )
# 添加数据
qr.add_data('你好')
# 生成二维码
qr.make(fit=True)
img = qr.make_image()
# image.save(r"C:\Desktop\s.png")
img.show()
