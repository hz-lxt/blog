BeautifulSoup模块的使用

1 参数介绍

html_doc = """
<html><head><title>The Dormouse's story</title></head>
<body>
<p id="my_p" class="title">hello<b id="bbb" class="boldest">The Dormouse's story</b>
</p>

<p class="story">Once upon a time there were three little sisters; and their names were
<a href="http://example.com/elsie" class="sister" id="link1">Elsie</a><a href="http://example.com/lacie" class="sister" id="link2">Lacie</a> and
<a href="http://example.com/tillie" class="sister" id="link3">Tillie</a>;
and they lived at the bottom of a well.</p>

<p class="story">...</p>
"""
soup = BeautifulSoup(html_doc,'lxml')

# 第一个参数是要解析的html页面，通常是我们通过请求得到的页面数据
# 第二个参数是解析的解析器选择
# python内置的html.parser
# 其他解释器：lxml（通常使用），xml，html5lib

文档：https://www.crummy.com/software/BeautifulSoup/bs4/doc/index.zh.html

2 基本使用

# 1 标签查找
soup = BeautifulSoup(html_doc,'lxml')
# 如果有多个只会找到第一个
res = soup.p
res = soup.p.b

# 2 标签属性
res = soup.p.b.attrs
&#123;'id': 'bbb', 'class': ['boldest']&#125;

# 3 获取文本内容
res = soup.p.string # 当p下只有文本取到，否则为None
res = soup.p.strings # 获取全部文本得到一个生成器对象
res = soup.p.text # 取到标签下所有的文本内容拼接起来

# 4 嵌套选择
# 可以让选择更加准确
res = soup.p.b.string

# 5 子节点、子孙节点
res = soup.p.contents # 获得p内所有子节点
['hello', <b class="boldest" id="bbb">The Dormouse's story</b>, '\n']

res = soup.p.children # 得到一个迭代器，包含p内所有子节点
'''
子节点
hello
<b class="boldest" id="bbb">The Dormouse's story</b>
'''

res = soup.p.descendants # 得到一个生成器，包含p内所有子孙节点
'''
子孙节点
hello
<b class="boldest" id="bbb">The Dormouse's story</b>
The Dormouse's story
'''

# 6 父节点、祖节点
res = soup.p.parent # 获得p的父节点
res = soup.p.parents # 获得p的父节点，和父节点的父节点。。。的生成器

# 7 兄弟节点
res = soup.p.next_sibling # 下一个兄弟
res = soup.p.next_previous_sibling # 上一个兄弟

3 搜索文档树

3.1 五种过滤器
# 1 字符串
# 如果有多个取第一个
res = soup.find('b')
# 获取全部，只有一个也是放在列表中
res = soup.find_all('b')

# 2 正则表达式
# 获得所有以b开头的标签：body，b
res = soup.find_all(re.compile('^b'))

# 3 列表
# a标签，b标签都会取
res = soup.find_all(['a','b'])

# 4 布尔
# 会查找到所有标签，不包括文本节点
res = soup.find_all(True)

# 5 自定义过滤器
# 会把整个页面传给tag
def has_class_not_id(tag):
    return tag.attrs.get('class') and not tag.attrs.get('id')
res = soup.find_all(has_class_not_id)
3.2 find_all()
# name：可以放任意类型的过滤器

# keyword：放键值对
res = soup.find_all(id = re.compile('^my'))

# 如果有特殊参数不能通过k，v来表示
# 比如data-foo = value
soup.find_all(attrs=&#123;"data-foo":"value"&#125;)

# 按类名的时候，关键字是class_
# 查找由sister和sss类的a标签
res = soup.find_all('a',class_='sister sss') 

# limit参数
# 找前连个a标签
res = soup.find_all('a',limit=2) 

# recursive = False 会查找tag下的直接的子节点
res = soup.html.find_all(recursive = False)
3.3 css选择器
# 选择器

# 类
res = soup.select('.sister')
# 含有sister类下的b标签
res = soup.select('.title b')

# id
res = soup.select('#sister')

# 获取属性
res = soup.select('.title b')[0].attrs

# 获取内容
res = soup.select('.title b')[0].text
