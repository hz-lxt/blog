2 类装饰器

2.1 无参装饰器

function logClass(params:any){
    console.log(params)
    params.prototype.apiUrl = "xxx"
}
@logClass
class HttpClient{
    constructor(){

    }
}
var h = new HttpClient()
// 可能会报错，检测不到这个属性，是ide报的错不用管
console.log(h.apiUrl)

2.2 有参装饰器（装饰器工厂）

// 有参装饰器
function logClass(msg:string){
    return function(params:any){
        params.prototype.apiUrl = msg
    }
}
@logClass("hz")
class HttpClient{
    constructor(){

    }
}
var h = new HttpClient()
console.log(h.apiUrl)

3 属性装饰器

注意点：

只能有参，传入的俩参数

  • 参数1：对于静态成员来说就是类的构造函数，对于实例成员来说就是类的原形对象
  • 参数2：成员名字
function logPrototype(params:any){
    // 外层函数接受参数
    console.log(params)
    // 内层函数接受类，属性名
    return function(targat:any,attr:any){
        console.log(targat)
        console.log(attr)

        // 注意这里在修改的时候需要用[]而不能直接点，当然也可以用原形链
        targat[attr] = params
    }
} 
class HttpClient{
    @logPrototype("baidu")
    public url: string | undefined
    constructor(){

    }
}
var h = new HttpClient()
console.log(h.url)

4 方法装饰器

接受三个参数

  • 参数1：对于静态成员来说是类的构造函数，对于实例成员是类的原形
  • 成员名字
  • 成员的属性描述符
function logFunc(params:any){
    // 外层函数接受参数
    console.log(params)
    // 内层函数接受类，属性名
    return function(target:any,funcName:any,desc:any){
        console.log(target)
        console.log(funcName)
        console.log(desc)
        // 这跟类装饰器没啥区别，直接拿到类了就能随便改了
        // 但是这里没有修改成功，看下面
        target.run = function(){
            console.log("123")
        }

    }
} 
class HttpClient{
    public url: string | undefined
    constructor(){

    }
    @logFunc("run func")
    run(){
        console.log("run")
    }
}
var h = new HttpClient()
h.run()

修改方法

// 有参装饰器
function logFunc(params:any){
    // 外层函数接受参数
    // console.log(params)
    // 内层函数接受类，属性名
    return function(target:any,funcName:any,desc:any){
        // desc.value 就是原来的方法
        var oMethod = desc.value
        desc.value = function(...args:any[]){
            args = args.map((value)=>{
                return String(value)
            })
            // 调用原来的方法
            // oMethod()
            // 为原来的方法传参
            oMethod.apply(this,args)
        }

    }
} 
class HttpClient{
    public url: string | undefined
    constructor(){

    }
    @logFunc("run func")
    run(...args:any[]){
        console.log(args)
        console.log("run") 
    }
}
var h = new HttpClient()
h.run(123,"zaaa")
