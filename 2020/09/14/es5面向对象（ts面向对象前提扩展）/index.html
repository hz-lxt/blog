es5面向对象（ts面向对象前提扩展）

1 创建对象

定义，创建，原形链，静态方法

// es5 中定义类和对象
function Peopel(){
    this.name = "hz"
    this.age = 18
    this.run = function(){
        console.log("i am "+this.name)
    }
}

// 通过原形链增加类中的方法和属性,原形链上的属性和方法会被所有实例使用
Peopel.prototype.sex = "boy"

Peopel.prototype.work = function(name){
    console.log(name)
}

// 配置静态方法
Peopel.getInfo = function(){
    console.log("我是类的静态方法")
}

var p = new Peopel()
console.log(p.sex)
p.run()
p.work("xxx")
Peopel.getInfo()

2 继承

2.1 对象冒充实现继承

// es5 中定义类和对象
function Peopel(){
    this.name = "hz"
    this.age = 18
    this.run = function(){
        console.log("i am "+this.name)
    }
}

// 通过原形链增加类中的方法和属性,原形链上的属性和方法会被所有实例使用
Peopel.prototype.sex = "boy"

Peopel.prototype.work = function(name){
    console.log(name)
}
// 继承：原形链+对象冒充的组合继承模式
function Web(){
    Peopel.call(this) // 对象冒充实现继承
    // 可以调用父类中的属性和方法，除原形链的方法和属性外
}

var w = new Web()
w.run()

function Web(){
}

3.3 原形链+构造函数组合继承

// es5 中定义类和对象
function Peopel(name,age){
    this.name = name
    this.age = age
    this.run = function(){
        console.log("i am "+this.name)
    }
}

// 通过原形链增加类中的方法和属性,原形链上的属性和方法会被所有实例使用
Peopel.prototype.sex = "boy"

Peopel.prototype.work = function(name){
    console.log(name)
}
// 继承：原形链+对象冒充的组合继承模式
function Web(name,age){
    Peopel.call(this,name,age)
}
// 第一种写法
Web.prototype = new Peopel()
// 第二种写法
Web.protorype = Peopel.protorype
var w = new Web("hz",123)
w.run()
w.work("hz")
