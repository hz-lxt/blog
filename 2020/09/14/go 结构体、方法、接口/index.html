1 结构体

定义：结构体是多个字段的集合

面向对象中类的定义是属性和方法的集合，结构体可以看做是属性的集合

1.1 结构体定义（创建命名结构体）

// type关键字 结构体名 struct关键字{}
type People struct {
    name string
    // 如果是大写开头表示可以在作为包导入中引用
    Age int
    sex string
}
func main() {
    // 两种定义方式，按位置必须全传，按关键字可以不按顺序少传
    peo := People{"hz",18,"男"}
    peo2 := People{name:"lxt",age:19}
}

补充：如果只定义不初始化内部只存在对应字段的零值

1.2 创建匿名结构体

匿名结构体通常在创建的时候就定义，一般只用一次，在整合多个数据类型的时候可以用，通常定义在方法内部或者结构体内部

dog := struct{
    name string
    age int
}{
    name:"xxx",
    age: 189,
}
fmt.Println(dog)

1.3 访问/修改结构体字段

// 直接.访问修改赋值
type People struct {
    name string
    age int
    sex string
}
func main() {
    peo := People{}
    peo.age = 19
    fmt.Println(peo.age)
}

1.4 结构体指针

// 指向结构体的指针，用法和结构体没差
type People struct {
    name string
    age int
    sex string
}
func main() {
    peo := &People{}
    // 原本必须要写(*peo).age go内部支持简写
    peo.age = 19
    fmt.Println(peo.age)
}

1.5 匿名字段

// 匿名字段本质就是把字段类型当名字一起使用
// 但是同时只能存在一种类型，就是不能写俩string
type People struct {
    name string
    age int
    sex string
    string
}
func main() {
    peo := &People{}
    peo.string = "xxx"
    fmt.Println(peo.string)
}

1.6 嵌套结构体

type People struct {
    name string
    age int
    sex string
    Hobby Hobby
}
type Hobby struct {
    hobbyId int
    hobbyName string
}

func main() {
    peo := People{}
    // 定义方式1
    //peo.Hobby.hobbyId = 1
    //peo.Hobby.hobbyName= "xxx"

    // 定义方式2
    peo.Hobby = Hobby{1,"xxx"}
    fmt.Println(peo.Hobby)
}

1.7 字段提升

在不重名的情况下可以把嵌套在内部的结构体字段提升到外层结构体来使用，如果重名就会优先使用外层结构体的字段名

type People struct {
    name string
    age int
    sex string
    // 必须使用匿名嵌套结构体才能提升
    Hobby
}
type Hobby struct {
    hobbyId int
    hobbyName string
}

func main() {
    peo := &People{}
    peo.Hobby = Hobby{1,"xxx"}
    fmt.Println(peo.hobbyName)
    fmt.Println(peo.hobbyId)
}

1.8 结构体相等性

type People struct {
    name string
    age int
}

func main() {
    peo1 := People{"hz",18}
    peo2 := People{"hz",18}
    // 两个字段稍微一点不同就是false
    fmt.Println(peo1 == peo2) // true

}
// 注意：结构体中只有全是值类型才能比较，如果存在引用类型比如切片，map这种引用类型就不能比较

2 方法

方法就是在定义函数之前加一个特殊的接收器类型，接收器可以是结构体或者非结构体，接收器内部可以访问这个方法

1.1 定义一个方法

type People struct {
    name string
    age int
}
// 为People结构体绑定了printName方法
func (p People)printName()  {
    fmt.Println(p.name)
}
func main() {
    // 初始化一个结构体对象
    peo := People{name: "hz"}
    peo.printName()
}

1.2 函数能实现同样的功能，为什么要方法？

  • go不是纯粹的面向对象的语言，go不支持类，方法是对应面向对象中对象的方法
  • 一个方法可以绑定给多个不同的结构体，但是一个函数不能接受不同的结构体作为参数

1.3 指针接收器与值接收器

如果是在方法中

有值接收器，那它可以接收值接收器和指针接收器

有指针接收器，那它可以接收值接收器和指针接收器

总结：不管方法的接收器是什么，都可以用值或者指针去接受，两种接收器的区别就是：值接收器是值传递，指针接收器是引用传递

type People struct {
    name string
    age int
}

func (p People)printName()  {
    fmt.Println(p.name)
}
// 如果只是简单的值传递，那在内部改的值其实是一个copy的新值
// 值接收器
func (p People)changeName()  {
    p.name = "lxt"
}

// 如果需要改动真的生效，就需要用到指针接收器
// 指针接收器（全都用这个就行了）
func (p *People)changeName2()  {
    p.name = "lxt"
}

func main() {
    peo := People{name: "hz"}
    //peo.changeName()  // 没有改原值
    peo.changeName2()
    peo.printName()
}

1.4 非结构体方法

如果要对非结构体绑定方法，那这个非结构体必须是我们自己定义的类型或者重命名的类型

type MyInt int
// 注意在运算中类型必须一致才能运算
// int不能传入结构体，必须重命名
func (i MyInt)add(x MyInt)  {
    fmt.Println(i+x)
}
func main() {
    var i MyInt = 5
    i.add(6)
}

3 接口

接口就是一系列方法的集合，类似于面向对象中的多态，go语言的接口也支持鸭子类型

面向对象领域中，接口是指定义和规范一个子类的行为

3.1 接口简单实现（鸭子类型）（非侵入式）

package main

import "fmt"

type Duck interface {
    speak()
    eat()
}
type TDuck struct {
    name string
    age int
}
type DDuck struct {
    name string
    age int
    salary int
}

func (t TDuck)speak()  {
    fmt.Println("TDUCK",t.name)
}
func (d DDuck)speak()  {
    fmt.Println("DDUCK",d.name)
}
func (t TDuck)eat()  {
    fmt.Println("T EAT",t.age)
}
func (d DDuck)eat()  {
    fmt.Println("D EAT",d.age)
}
func main() {
    // 让Tduck和Dduck都实现Duck接口
    // 方式一：通过鸭子类型，写同样的方法
    t :=TDuck{"hz",18}
    t.speak()

    d:=DDuck{"lxt",19,2000}
    d.eat()
}

3.2 侵入式接口

type Duck interface {
    speak()
    eat()
}
func (t TDuck)speak()  {
    fmt.Println("TDUCK",t.name)
}
func (d DDuck)speak()  {
    fmt.Println("DDUCK",d.name)
}
func (t TDuck)eat()  {
    fmt.Println("T EAT",t.age)
}
func (d DDuck)eat()  {
    fmt.Println("D EAT",d.age)
}

type TDuck struct {
    name string
    age int
}
type DDuck struct {
    name string
    age int
    salary int
}


func main() {
    // 让Tduck和Dduck都实现Duck接口
    // 方式二：侵入式接口，接口方法没有，报错
    var duck Duck
    t:=TDuck{"hz",18}

    duck=t
    duck.eat()
    d:=DDuck{"lxt",19,2000}
    duck = d
    duck.eat()
}

3.3 接口的内部表示

我们可以把接口看作内部的一个元组 (type, value)。
type 是接口底层的具体类型（Concrete Type），而 value 是具体类型的值。

3.4 类型断言

package main

import "fmt"

type Duck interface {
    speak()
    eat()
}
func (t TDuck)speak()  {
    fmt.Println("TDUCK",t.name)
}
func (d DDuck)speak()  {
    fmt.Println("DDUCK",d.name)
}
func (t TDuck)eat()  {
    fmt.Println("T EAT",t.age)
}
func (d DDuck)eat()  {
    fmt.Println("D EAT",d.age)
}

type TDuck struct {
    name string
    age int
}
type DDuck struct {
    name string
    age int
    salary int
}


func main() {
    // 断言类型
    // 把接口类型转换成结构体类型

    // 定义了一个接口类型
    var tduck Duck = TDuck{"hz",18}
    // 断言这个值的类型，v为值，ok为判断的bool
    // 如果断言失败，ok为false v为空值
    v,ok := tduck.(TDuck)
    fmt.Println(v)
    fmt.Println(ok)
}

3.5 空接口

package main

import "fmt"

type Empty interface {

}
func main() {
    // 定义一个空接口
    // 所有数据类型都实现了空接口，所以空接口可以接受任意类型的值
    var e Empty
    a := 123
    e = a
    // 虽然可以接受任意类型，但是有一个缺点就是本身没有任何方法
    // 所以要使用原来值类型的方法就必须类型断言回去，再使用
    fmt.Println(e)
}

同理，集合中也可以放入空接口来接收任意值

4 自定义集合类型

package main

import "fmt"

//定义MySet类型
type MySet map[interface{}]bool
//判断元素是否存在
func (m MySet) isExist(a interface{}) bool {
    return m[a]
}
//返回set长度
func (m MySet) len() int {
    return len(m)
}
//设置值
func (m MySet) set(a interface{}) {
    m[a] = true
}
//删除值
func (m MySet) delete(a interface{}) {
    delete(m, a)
}
//测试代码
func main() {
    //创建一个set
    var a MySet = make(MySet)
    //相当于
    //var a MySet = make(map[interface{}]bool)
    //打印set的长度
    //fmt.Println(a.len())
    //放入一个值
    a.set(1)
    //放入一个相同值
    a.set(1)
    a.set("lqz")
    a.set("lqz")
    a.set("lqz")
    a.set("lqz")
    a.set("lqz")
    a.set("lqz")
    //打印长度，还是1
    //fmt.Println(a.len())
    //判断1是否存在
    //fmt.Println(a.isExist(2))
    ////删除1
    a.delete(1)
    ////判断1是否存在
    fmt.Println(a.isExist(1))
    fmt.Println(a.len())

    for i,_:=range a{
        fmt.Println(i)
    }
}
