3 可选参数

// 可选参数
// ts中默认必须按照形参传实参，如果设置某个参数可以不传就需要配置
function prinyMsg(name:string,age?:number):string{
    if(age){
        return `${name}   ${age}`
    }else{
        return `${name} age保密`
    }
}

var res = prinyMsg("hz")
var res1 = prinyMsg("lxt",18)
console.log(res)
console.log(res1)

注意：可选参数必须放在形参的最后面，不然会报错

4 默认参数

es5中不能设置默认参数，es6和ts中都可以设置默认参数

注意：默认参数也必须写在最后，否则没有效果

// 默认参数

function prinyMsg2(name:string,age:number=19):string{
    if(age){
        return `${name}   ${age}`
    }else{
        return `${name} age保密`
    }
}
var res = prinyMsg2("hz")
var res = prinyMsg2("hz",20)

5 剩余参数（三点运算符）

function add2(...result:number[]):number{
    var sum = 0
    for(var i =0;i<result.length;i++){
        sum+=result[i]
    }
    return sum
}

var res = add2(1,2,3,4,5)
console.log(res)

6 函数重载

es5中出现同名函数，下面的会替换上面的


// 方法重载
// 如果我只写两个getInfo就会报错，必须要有重载函数
function getInfo(name:string):string;
function getInfo(age:number):string;
function getInfo(str:any):any{
    if(typeof str === "string"){
        return "我是"+ str
    }else if(typeof str === "number"){
        return "我的年龄是" + str
    }
}
// 且这里传参必须传能找到的，否则报错
var res = getInfo(18)
console.log(res)
