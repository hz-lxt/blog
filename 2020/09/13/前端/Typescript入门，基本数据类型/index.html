1 基本介绍

ts是js的超集，是目前前端必备的基础，包含了es5，es6的语法

也可以集成到vue中

安装：cnpm install -g typescript

查看版本：tsc -v

语言类型：静态弱类型

无法直接在浏览器运行，需要编译成js代码

2 基本使用

2.1 如何配置自动编译

使用idev：vscode

在vscode中在项目下cmd窗口输入tsc --init就会自动生成一个.json的配置文件

修改配置文件中

// 默认是注释的解开注释即可
"outDir": "./js"
// 会把编译完成的ts代码放到js目录下的文件中

然后点击终端–typescript–监视配置文件

如果报错9成是因为tsc权限不足

解决方式：以管理员方式打开cmd

输入set-ExecutionPolicy RemoteSigned

再输入A或者Y即可

3 数据类型

3.1 数据类型定义

// 布尔类型 boolean
let boo:boolean = true

// let关键字 变量名:变量类型 = 值

3.2 数字类型(number)

let num:number = 123
let num:number = 123.3

// 数值类型包括整形和浮点型

3.3 字符类型（string）

let str:string = "hz"

3.4 数组类型（array）

// 数组在定义的时候必须指定内部的类型
// 第一种定义方式
let test_arry1:number[] = [1,3,456]
let test_arry2:string[] = ["hz","lxt"] // 推荐用双引号

// 第二种定义方式
let test_arry3:Array<number> = [1,2,3,4]

// 第三种定义方式
// 通过any类型就可以放任意类型
let arr:any[] = ['123',123,true]

3.5 元组类型（tuple）

// 可以定义内部类型的数组，必须一一对应
// 元组类型（数组的一种）
let test_arry4:[number,string,string[]] = [1,"hz",["hz","nb"]]

3.6 枚举类型（enum）

// 枚举类型，相当于创建一种新的类型，通过标识符和数值的对应来表示值类
// 用于标识一种状态
enum Flag {success=1,error=2}
var f:Flag = Flag.success 
console.log(f) // 1

console.log(Flag.error) //2 

// 如果枚举类型没有定义值，打印的就是索引值
enum Color {red,blue=5,green}
var c:Color = Color.red
console.log(c) // 0
console.log(Color.blue) // 5
console.log(Color.green) // 6 如果上一个对应值不是索引，就会按照上一个值递增1

3.7 任意类型（any）

// 任意类型 any
let data:any = 123

data = '123'

// 任意类型的用途
// 当我们需要用变量接受一个对象时，js中是可以直接接受
// 必须指定any
let oBox:any = document.getElementById('box')
oBox.style.color = 'red'

3.8 其他类型（never类型）的子类型

代表从不会出现的值

以为这never类型的变量只能被类型所赋值

// 其他类型（never类型）的子类型
//undefined
// num此时只能被赋值undefined
var num:undefined
console.log(num)

// 如果被赋值就是number类型，没赋值就是undefined类型
var num:number | undefined
console.log(num)

// null
// undefined可以不赋值是因为他不仅可以作为类型，也可以作为值，而null不一样，如果要定义null类型就必须赋值null值
var num1:null = null
console.log(num1)

// 综合运用
// 如果我们对一个值的类型不能确定，不仅可以用any，也可以用|分隔可能出现的类型
let num:number | null | undefined
console.log(num) // undefined
num = null
console.log(num)// null
num = 123
console.log(num) // 123

3.9 void表示没有任何类型

一般用于定义方法的时候没有任何返回值

// 表示这个方法没有返回值
function run1():void{
    console.log('456')
}
run1()

// 如果一个方法有返回值，就指定返回值类型
function run1():number{
    return 123
}
console.log(run1())
