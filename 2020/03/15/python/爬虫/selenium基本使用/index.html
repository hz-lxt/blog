一、安装

  • 模块安装：pip install selenium
  • 驱动安装，安装和浏览器适配的驱动exe，放到解释器的script文件夹中（http://npm.taobao.org/mirrors/chromedriver/）

二、基本使用

# 案例：打开百度输入搜索海贼王

from selenium import webdriver
from selenium.webdriver.common.keys import Keys
import time

# 得到一个谷歌浏览器驱动对象
browser = webdriver.Chrome()
browser.implicitly_wait(5) # 隐式等待：页面中查找标签时等待，如果超出时间还没找到就报错
browser.get("https://www.baidu.com/")

input_tag = browser.find_element_by_id('kw')
input_tag.send_keys('海贼王')
# input_tag.send_keys(Keys.ENTER)

commit_tag = browser.find_element_by_id('su')
commit_tag.click()
time.sleep(2)
browser.close()

所有方法

# ##############选择器（find系列）
# ===============所有方法===================
# 1、find_element_by_id   # 通过id查找控件
# 2、find_element_by_link_text  # 通过a标签内容找
# 3、find_element_by_partial_link_text  # 通过a标签内容找，模糊匹配
# 4、find_element_by_tag_name   # 标签名
# 5、find_element_by_class_name  # 类名
# 6、find_element_by_name      # name属性
# 7、find_element_by_css_selector  # 通过css选择器
# 8、find_element_by_xpath       # 通过xpaht选择器

css选择中
-response.css('.link-title::text')# 取文本
-response.css('.link-title::attr(href)')# 取标签
# 以上全都可以改成find_elements_xxx获取多个

获取属性

# 重点
# tag.get_attribute('href')  # 找当前控件 的href属性对的值
# tag.text   # 获取文本内容
