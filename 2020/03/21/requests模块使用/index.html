1 发送请求

2 请求地址中携带数据

# url的编码和解码
from urllib.parse import urlencode,unquote

header = {
     'user-agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/84.0.4147.105 Safari/537.36',
 }
res=requests.get('https://www.baidu.com/s',headers=header,params={'wd':'美女'})

3 请求中带cookie

# 方式一：在headers中携带
header = {
    'cookie':'name=hz'
}
r = requests.get('http://127.0.0.1:8000/test/',headers=header)


# 方式二：添加cookies参数
r = requests.get('http://127.0.0.1:8000/test/',cookies={'name':'hz'})

4 post请求发送url和json格式数据

# url格式
r = requests.post('http://127.0.0.1:8000/test/',data={'name':'hz'})

# post请求发送json格式数据
r = requests.post('http://127.0.0.1:8000/test/',json={'name':'hz'})

5 session函数自动处理携带cookie

session = requests.session()

r = session.get('http://127.0.0.1:8000/login/')
print(r.cookies)
r2 = session.post('http://127.0.0.1:8000/order/')

6 response对象的方法

respone.text # 响应的文本信息
print(respone.content)  # 响应体的二进制
print(respone.status_code)  # 响应状态码
print(respone.headers)    # 响应头
print(respone.cookies)   # cookie
print(respone.cookies.get_dict()) #  把cookie转成字典
print(respone.cookies.items())  # key和value
print(respone.url)        # 请求的url
print(respone.history)   #[]放重定向之前的地址
print(respone.encoding)  # 响应的编码方式
respone.iter_content()  # 图片，视频，大文件，一点一点循环取出来

7 编码问题

res=requests.get('http://www.autohome.com/news')
# 一旦打印出来出现乱码问题

# 方式一：手动设定编码格式
res.encoding='gb2312'

# 方式二
res.encoding=res.apparent_encoding
print(res.text)

8 json解析

import json
respone=requests.post('http://127.0.0.1:8000/index/',data={'name':'lqz'})
print(type(respone.text))  # 响应的文本
print(json.loads(respone.text))
print(respone.json())  # 相当于上面那句话
