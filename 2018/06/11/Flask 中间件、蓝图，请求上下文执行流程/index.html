from flask import Flask,make_response,session,request,flash,get_flashed_messages,redirect

app = Flask(__name__)


class Md(object):
    def __init__(self,old_wsgi_app):
        self.old_wsgi_app = old_wsgi_app

    def __call__(self,  environ, start_response):
        print('开始之前')
        ret = self.old_wsgi_app(environ, start_response)
        import time
        time.sleep(3)
        print('结束之后')
        return ret

@app.route('/',methods=['GET'])

四、partal偏函数（扩展）

# 偏函数的第一个参数可以放一个函数的内存地址
# 后续参数为这个函数的参数，如果传的少了，这个偏函数的返回结果是一个新的函数
# 这个函数已经通过偏函数为前几个参数传了值，后续只要补上后面的参数即可
from functools import partial
def test(a,b,c,d):
    return a+b+c+d

tes=partial(test,1,2)
print(tes(3,4))

五、threading.local

flask实现每次请求都是不同的request但是这个request是全局变量的原因

当一个请求来的时候，uwsgi为我们开启了一个线程/协程

此时request作为全局变量是放在解释器中的，也就是如果多线程去操作这个数据的话必定会发生数据错乱，此时我们要保证数据不乱，同时不能用锁来影响我们的并发

try:
    from greenlet import getcurrent as get_ident
except Exception as e:
    from threading import get_ident
# 上面的代码保证了线程和协程都能有独自的id号
# 因为如果在线程里开设了多个协程，只获取线程id就是一样的
from threading import Thread
import time
class Local(object):
    def __init__(self):
        object.__setattr__(self,'storage',&#123;&#125;)
    def __setattr__(self, k, v):
        ident = get_ident()
        if ident in self.storage:
            # storage：｛线程id：app对象，。。｝
             self.storage[ident][k] = v
         else:
             self.storage[ident] = &#123;k: v&#125;
    def __getattr__(self, k):
        ident = get_ident()
        return self.storage[ident][k]
obj = Local()
def task(arg):
    obj.val = arg
    # obj.xxx = arg

    print(obj.val)
for i in range(10):
    t = Thread(target=task,args=(i,))
    t.start()

六、请求上下文执行流程

# 第一步 app.run 找到核心逻辑代码
# 最终调用了werkzeug的run_simple，然后传了一堆配置参数，这个函数会调用执行app
# 也就是会执行到Flask类的__call__方法
# 这里又调用了wsgi_app方法，这就是整个请求流程的核心代码

# 第二步 解析wsgi_app方法
# 这一步返回了RequestContext对象，内部封装了本次请求request，session等信息
ctx = self.request_context(environ)

# 上面已经知道了这个ctx就是RequestContext对象
ctx.push()
# 这个push的最后为我们设置了真正的session
# 在这之上调用了_request_ctx_stack.push
_request_ctx_stack.push(self)

# _request_ctx_stack是Flask创建出来就产生了的顶级变量，总共有6个
# 它是由LocalStack类实例化得到的
_request_ctx_stack = LocalStack()
# 也就是我们上面找的push实际上是LocalStack的函数
    def push(self, obj):
        rv = getattr(self._local, "stack", None)
        if rv is None:
            # _local是flask自带的local对象
            # 他的特点是不同线程放的值只有自己这次请求能用
            self._local.stack = rv = []
        # obj是ctx对象
        rv.append(obj)
        return rv
# 继续解析wsgi_app
# 这一步从字面上意思就是把请求对象转化成响应对象，内部应该就是请求流程的整体逻辑了
response = self.full_dispatch_request()

  def full_dispatch_request(self):
        # 这是第一次请求时触发的被before_first装饰的函数
        # 内部就是执行了全部的这些函数然后把标志第一次的参数改成True
        self.try_trigger_before_first_request_functions()
        try:
            # 信号相关
            request_started.send(self)
            # 被before_request装饰的函数，所以他的返回值已经被限定死了
            rv = self.preprocess_request()
            if rv is None:
                # 这里真正的根据路由执行视图函数
                rv = self.dispatch_request()
        except Exception as e:
            rv = self.handle_user_exception(e)
        # rv是我们的响应对象，这里通过make_response处理了保证字符串也能转成response对象
        # 然后它还会调用视图中被after_reuqest装饰的函数，内部规定了传值和接收值，所以我们的请求扩展要那么写
        # 还会把session放到rv中，也就是在请求扩展中是可以拿到session的
        return self.finalize_request(rv)
# 继续解析wsgi_app
# 上面已经拿到响应对象了，也就是其实到这里请求的所有流程已经结束了，在这最后会把我们的ctx对象从local中移除
ctx.auto_pop(error)

# 总结：从run方法开始，调用了run就是调用了werkzeug的run_simple
# run_simple配置了很多参数调用了app，也就是调用了Flask类的__call__方法
# 内部又调用了wsgi_app方法，这是Flask请求响应的核心逻辑
# 先会生成一个ctx对象，这是本次请求的所有信息（request，session，app）通过线程号作为key，信息作为value放到local字典中，所以每次取的时候都是唯一的
# 然后去执行了请求扩展的一些函数，比如befor_request
# 请求处理完成之后就是获取响应对象，就是执行了视图函数，返回的响应对象需要传给响应扩展函数
# 就是那些被after_request装饰的函数，执行完成后获得真正的响应对象，返回给前端
# 最后要把ctx对象从原来的字典中移除，因为本次请求已经完成

如何全局拿到唯一的request对象

# request是视图中的全局变量，但是却唯一
# 这里使用偏函数包装起来，此时并没有执行
# 我们可以打印request对象的方法，那么LocalProxy内部必定重写了__str__方法
request = LocalProxy(partial(_lookup_req_object, "request"))

# 打印的返回值是obj，这里可以猜测obj就是request对象
obj = self._get_current_object()

    def _get_current_object(self):
        # 去找LocalProxy内部的__local
        if not hasattr(self.__local, "__release_local__"):
            return self.__local()
# 这里用了封装，所以我们可以在__init__中找到_LocalProxy__local
    def __init__(self, local, name=None):
        object.__setattr__(self, "_LocalProxy__local", local)
        object.__setattr__(self, "__name__", name)
        if callable(local) and not hasattr(local, "__release_local__"):
            object.__setattr__(self, "__wrapped__", local)
# local是我们在类实例化的时候通过偏函数传进来的
# 这时候再看这一句实例化的代码就能理解了，此时实例化的过程中传进来的偏函数内存地址
# 在__init__中被调用执行获得了执行的记过，而这个结果返回给了__str__让他去打印
# 之前就说了视图中打印的request对象是唯一的，所以现在只要知道_lookup_req_object是什么就ok了
LocalProxy(partial(_lookup_req_object, "request"))

def _lookup_req_object(name):
    # 去local中拿到ctx对象（上面就解析过，ctx包含了request，app，session）
    top = _request_ctx_stack.top
    if top is None:
        raise RuntimeError(_request_ctx_err_msg)
    # name是偏函数传进来的request字符串，通过反射获取了指定ctx对象的request
    return getattr(top, name)

# 至此我们知道了request是如何拿出来唯一的，还有一个小点就是如何拿到request内部的其他对象，request取值都是通过点
# 那就必定重写了__getattr__
    def __getattr__(self, name):
        if name == "__members__":
            return dir(self._get_current_object())
        # 很清晰，从传过来的name来到request对象中反射
        # 比如request.method
        return getattr(self._get_current_object(), name)

# 总结：Flask中的request虽然是个全局变量但是每次请求拿到的是唯一的，在Flask启动的时候就会创建LocalProxy对象
# 它的实例化需要传一个参数，这里用的是偏函数，先从一个方法中通过request在ctx对象中的反射获取唯一的request对象
# 然后在视图中打印的时候就会触发LocalProxy对象的__str__方法
# 此时会从init中拿到实例化时被封装传进来的local，并执行，得到的就是唯一的那个request对象了
# 然后在点取值，比如request.method的时候会触发__getattr__方法，同样是去拿到request对象从内部反射出结果返回
